วิธีสร้าง Google Apps Script

  1. 1 เปิด Google Sheets สร้างไฟล์ใหม่ แล้วตั้งชื่อ Sheet 4 แท็บ ว่า Members, Profiles, LoginLogs, Notifications
  2. 2 ไปที่เมนู ส่วนขยาย → Apps Script แล้ววางโค้ดด้านล่างลงไป

  4. 4 Copy Web App URL มาวางในช่องด้านบน
Google Apps Script Code 
// ============================================================
//  Google Apps Script สำหรับ MemberApp
//  วางในไฟล์ Code.gs แล้ว Deploy เป็น Web App
// ============================================================

const SHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const action = data.action;
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let result = { success: false };

  try {
    if (action === 'register') {
      // บันทึกข้อมูลสมัครสมาชิก
      const sheet = ss.getSheetByName('Members') || ss.insertSheet('Members');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Timestamp','FirstName','LastName','Email','Phone','DOB','PasswordHash']);
      }
      // ⚠️ ในระบบจริงควร hash รหัสผ่านฝั่ง Server ก่อนบันทึก
      sheet.appendRow([
        new Date().toLocaleString('th-TH'),
        data.firstName, data.lastName, data.email,
        data.phone, data.dob, data.password
      ]);
      result = { success: true, message: 'สมัครสมาชิกสำเร็จ' };

    } else if (action === 'login_log') {
      // บันทึก Login Log
      const sheet = ss.getSheetByName('LoginLogs') || ss.insertSheet('LoginLogs');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Timestamp','Email','Status','UserAgent']);
      }
      sheet.appendRow([
        new Date().toLocaleString('th-TH'),
        data.email, data.status, data.userAgent
      ]);
      result = { success: true };

    } else if (action === 'save_notifications') {
      // บันทึกการตั้งค่าการแจ้งเตือน
      const sheet = ss.getSheetByName('Notifications') || ss.insertSheet('Notifications');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Email','SystemEmail','ActivityAlert','Promotions','UpdatedAt']);
      }
      // Upsert: ค้นหาอีเมลเดิมก่อน
      const vals = sheet.getDataRange().getValues();
      let found = false;
      for (let i = 1; i < vals.length; i++) {
        if (vals[i][0] === data.email) {
          sheet.getRange(i + 1, 1, 1, 5).setValues([[
            data.email, data.systemEmail, data.activityAlert,
            data.promotions, new Date().toLocaleString('th-TH')
          ]]);
          found = true; break;
        }
      }
      if (!found) {
        sheet.appendRow([
          data.email, data.systemEmail, data.activityAlert,
          data.promotions, new Date().toLocaleString('th-TH')
        ]);
      }
      result = { success: true, message: 'บันทึกการแจ้งเตือนสำเร็จ' };

    } else if (action === 'save_profile') {
      // บันทึก/อัปเดตโปรไฟล์
      const sheet = ss.getSheetByName('Profiles') || ss.insertSheet('Profiles');
      if (sheet.getLastRow() === 0) {
        sheet.appendRow(['Email','FirstName','LastName','Phone','DOB','UpdatedAt']);
      }
      // ค้นหาแถวที่มีอีเมลนี้อยู่แล้ว
      const values = sheet.getDataRange().getValues();
      let found = false;
      for (let i = 1; i < values.length; i++) {
        if (values[i][0] === data.email) {
          sheet.getRange(i + 1, 1, 1, 6).setValues([[
            data.email, data.firstName, data.lastName,
            data.phone, data.dob, new Date().toLocaleString('th-TH')
          ]]);
          found = true;
          break;
        }
      }
      if (!found) {
        sheet.appendRow([
          data.email, data.firstName, data.lastName,
          data.phone, data.dob, new Date().toLocaleString('th-TH')
        ]);
      }
      result = { success: true, message: 'บันทึกโปรไฟล์สำเร็จ' };
    }
  } catch (err) {
    result = { success: false, message: err.toString() };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const action = e.parameter.action;
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let result = {};

  if (action === 'stats') {
    const members = ss.getSheetByName('Members');
    const logs = ss.getSheetByName('LoginLogs');
    const memberCount = members ? Math.max(0, members.getLastRow() - 1) : 0;

    let todayLogins = 0;
    if (logs && logs.getLastRow() > 1) {
      const today = new Date().toLocaleDateString('th-TH');
      const logData = logs.getDataRange().getValues();
      for (let i = 1; i < logData.length; i++) {
        const rowDate = new Date(logData[i][0]).toLocaleDateString('th-TH');
        if (rowDate === today && logData[i][2] === 'success') todayLogins++;
      }
    }
    result = { success: true, memberCount, todayLogins };

  } else if (action === 'recent_logs') {
    const logs = ss.getSheetByName('LoginLogs');
    const rows = [];
    if (logs && logs.getLastRow() > 1) {
      const data = logs.getDataRange().getValues();
      const recent = data.slice(1).reverse().slice(0, 10);
      recent.forEach(r => rows.push({
        timestamp: r[0], email: r[1], status: r[2]
      }));
    }
    result = { success: true, logs: rows };
  }

  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
